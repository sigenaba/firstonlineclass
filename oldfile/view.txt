SELECT a.id AS id, a.sold_date AS sold_date, b.status AS status, c.item_description AS cpu,
 d.item_description AS hdd, e.item_description AS ram, f.item_description AS mobo, g.item_description AS os, h.buyer_name AS buyername
FROM system_unit a
INNER JOIN system_status b ON b.id = a.status_id
INNER JOIN item_masterfile c ON c.id = a.cpu_id
INNER JOIN item_masterfile d ON d.id = a.hdd_id
INNER JOIN item_masterfile e ON e.id = a.ram_id
INNER JOIN item_masterfile f ON f.id = a.mobo_id
INNER JOIN item_masterfile g ON g.id = a.os_id
LEFT JOIN buyer h ON h.id = a.buyer_id
